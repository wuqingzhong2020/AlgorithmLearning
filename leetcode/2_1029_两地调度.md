题型解法 ： [贪心](..\leetcode.md#贪心题)

## 思路 
```
假设2n个人都是去A地， 现在把n个人改去B地， 
要想费用最少， 那么选取改到B地的费用用最少n个人， 
改到B的费用可能为负数
```

## c++
```c++
class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int sum = 0; // 不会溢出 10^2 * 10^3 = 10^5 
        std::vector<int> arr(costs.size());
        for (int i = 0; i < costs.size(); i++) {
            sum += costs[i][0]; // 统计都去A地的费用
            arr[i] = costs[i][1] - costs[i][0]; // 从A地改去B地，要多花费的费用
        }

        std::ranges::sort(arr);
        for (int i = 0; i < costs.size()/2; i++) {
            sum += arr[i];
        }
        return sum;
    }
};
```

## python3
```python
class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        # [[1,1], [2,2]] 数据结构
        sum = 0
        # arr = list()
        # 指定元素个数
        arr = [0 for _ in range(len(costs))]
        for i in range(len(costs)):
            sum += costs[i][0]
            # arr.append(costs[i][1] - costs[i][0])
            arr[i] = costs[i][1] - costs[i][0]
        arr.sort()
        for i in range(len(arr) // 2):
            sum += arr[i]

        return sum
```

impl Solution {
    pub fn two_city_sched_cost(costs: Vec<Vec<i32>>) -> i32 {
        let mut sum: i32 = 0;
        let n = costs.len();
        let mut cost2b  = Vec::with_capacity(n);
        // 没智能模式好难写
        for item in &costs {
            sum += item[0];
            cost2b.push(item[1] - item[0])
        }

        cost2b.sort();

        for i in 0..(n/2) {
            sum += cost2b[i];
        }
        sum
    }
}
